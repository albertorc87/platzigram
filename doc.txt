# CREAR ENTORNO VIRTUAL
## Windows
> python -m venv .env
# Activar entono virtual
> .env/Scripts/activate
# Desactivar entorno virtual
> .env/Scripts/deactivate

## Ubuntu
$ python3 -m venv .env
# Activar entono virtual
$ source .env/bin/activate
# Desactivar entorno virtual
$ deactivate

## Mac
python3 -m venv .env
# Activar entono virtual
source .env/bin/activate
# Desactivar entorno virtual
deactivate
